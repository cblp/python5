  product = 1

      product *= n
  print("The product is",
        product)

  numbers = [2, 4, 6, 8]  product <- 1  for n in numbers:
      product *= n
  print("The product is",
        product)
   
  The product is 384

  ...     for i in range(3)
  ... ]

  >>> adders = [
  ...     x -> x + i
  ...     for i in range(3)
  ... ]

  >>> [f(0) for f in adders]

  [0, 1, 2]
